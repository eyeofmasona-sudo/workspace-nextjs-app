   259→// ─── Pixel Office Layout (40×36) ─────────────────────────────
   260→const COLS = 40;

   263→function buildTiles(): TileTypeVal[] {
   264→  const W = 0 as TileTypeVal;
   265→  const F1 = 1 as TileTypeVal;
   266→  const F2 = 2 as TileTypeVal;
   267→  const F3 = 3 as TileTypeVal;
   268→  const F4 = 4 as TileTypeVal;
   269→  const F5 = 5 as TileTypeVal;
   270→  const F6 = 6 as TileTypeVal;
   271→  const F7 = 7 as TileTypeVal;
   272→  const F8 = 8 as TileTypeVal;
   273→  const F9 = 9 as TileTypeVal;
   274→  const F10 = 10 as TileTypeVal;
   275→
   276→  const tiles: TileTypeVal[] = [];
   277→  for (let r = 0; r < ROWS; r++) {
   278→    for (let c = 0; c < COLS; c++) {
   279→      if (r === 0 || r === ROWS - 1 || c === 0 || c === COLS - 1) { tiles.push(W); continue; }
   280→      if (r === 10) { tiles.push((c >= 5 && c <= 6) || (c >= 18 && c <= 19) || (c >= 31 && c <= 32) ? F8 : W); continue; }
   281→      if (r === 20) { tiles.push((c >= 5 && c <= 6) || (c >= 18 && c <= 19) || (c >= 31 && c <= 32) ? F8 : W); continue; }
   282→      if (r === 30) { tiles.push((c >= 8 && c <= 9) || (c >= 28 && c <= 29) ? F8 : W); continue; }
   283→      if (r >= 1 && r <= 9) {
   284→        if (c === 13) { tiles.push(r >= 4 && r <= 5 ? F8 : W); continue; }
   285→        if (c === 25) { tiles.push(r >= 4 && r <= 5 ? F8 : W); continue; }
   286→        if (c >= 1 && c <= 12) { tiles.push(F1); continue; }
   287→        if (c >= 14 && c <= 24) { tiles.push(F2); continue; }
   288→        if (c >= 26 && c <= 38) { tiles.push(F3); continue; }
   289→        tiles.push(W); continue;
   290→      }
   291→      if (r >= 11 && r <= 19) {
   292→        if (c === 13) { tiles.push(r >= 14 && r <= 15 ? F8 : W); continue; }
   293→        if (c === 25) { tiles.push(r >= 14 && r <= 15 ? F8 : W); continue; }
   294→        if (c >= 1 && c <= 12) { tiles.push(F4); continue; }
   295→        if (c >= 14 && c <= 24) { tiles.push(F5); continue; }
   296→        if (c >= 26 && c <= 38) { tiles.push(F6); continue; }
   297→        tiles.push(W); continue;
   298→      }
   299→      if (r >= 21 && r <= 29) {
   300→        if (c === 13) { tiles.push(r >= 24 && r <= 25 ? F8 : W); continue; }
   301→        if (c === 25) { tiles.push(r >= 24 && r <= 25 ? F8 : W); continue; }
   302→        if (c >= 1 && c <= 12) { tiles.push(F9); continue; }
   303→        if (c >= 14 && c <= 24) { tiles.push(F10); continue; }
   304→        if (c >= 26 && c <= 38) { tiles.push(F7); continue; }
   305→        tiles.push(W); continue;
   306→      }
   307→      tiles.push(F7);
   308→    }
   309→  }
   310→  return tiles;
   311→}

   313→function buildTileColors(): Array<{ h: number; s: number; b: number; c: number } | null> {
   314→  const wallColor = { h: 214, s: 30, b: -100, c: -55 };
   315→  const cmdColor = { h: 30, s: 50, b: -43, c: -88 };
   316→  const meetColor = { h: 210, s: 35, b: -30, c: -75 };
   317→  const designColor = { h: 330, s: 40, b: -35, c: -80 };
   318→  const devColor = { h: 150, s: 40, b: -40, c: -82 };
   319→  const serverColor = { h: 200, s: 45, b: -50, c: -70 };
   320→  const researchColor = { h: 35, s: 55, b: -38, c: -85 };
   321→  const loungeColor = { h: 20, s: 60, b: -35, c: -85 };
   322→  const doorColor = { h: 35, s: 25, b: 10, c: 0 };
   323→  const qaColor = { h: 345, s: 40, b: -40, c: -82 };
   324→  const opsColor = { h: 25, s: 50, b: -38, c: -80 };
   325→  const tiles = buildTiles();
   326→  const colors: Array<{ h: number; s: number; b: number; c: number } | null> = [];
   327→  for (let i = 0; i < tiles.length; i++) {
   328→    const tile = tiles[i];
   329→    if (tile === 0) { colors.push(wallColor); continue; }
   330→    if (tile === 8) { colors.push(doorColor); continue; }
   331→    if (tile === 1) { colors.push(cmdColor); continue; }
   332→    if (tile === 2) { colors.push(meetColor); continue; }
   333→    if (tile === 3) { colors.push(designColor); continue; }
   334→    if (tile === 4) { colors.push(devColor); continue; }
   335→    if (tile === 5) { colors.push(serverColor); continue; }
   336→    if (tile === 6) { colors.push(researchColor); continue; }
   337→    if (tile === 7) { colors.push(loungeColor); continue; }
   338→    if (tile === 9) { colors.push(qaColor); continue; }
   339→    if (tile === 10) { colors.push(opsColor); continue; }
   340→    colors.push(wallColor);
   341→  }
   342→  return colors;
   343→}

   545→// ─── Pixel Office Status Mapping ──────────────────────────────
   546→
   547→function mapAgentStatusToActive(status: string): boolean {
   548→  const s = status?.toLowerCase() ?? 'idle';
   549→  return !['offline', 'idle', 'done'].includes(s);
   550→}
   551→
   552→function mapAgentStatusToTool(status: string): string | null {
   553→  const s = status?.toLowerCase() ?? '';
   554→  if (s.includes('thinking') || s.includes('working')) return 'Write';
   555→  if (s.includes('review') || s.includes('reading') || s.includes('waiting_api')) return 'Read';
   556→  return null;
   557→}
   558→
   559→function mapAgentStatusToBubble(status: string): 'permission' | 'waiting' | null {
   560→  const s = status?.toLowerCase() ?? '';
   561→  if (s.includes('waiting_approval')) return 'permission';
   562→  if (s.includes('waiting_api')) return 'waiting';
   563→  return null;
   564→}
   565→
   566→function mapAgentStatusToBehavior(status: string): BehaviorState {
   567→  const s = status?.toLowerCase() ?? 'idle';
   568→  if (s.includes('working') || s.includes('thinking')) return BehaviorState.WORKING;
   569→  if (s.includes('reviewing')) return BehaviorState.MEETING;
   570→  if (s.includes('waiting_approval')) return BehaviorState.MEETING;
   571→  if (s.includes('waiting_api')) return BehaviorState.RESEARCH;
   572→  if (s.includes('done')) return BehaviorState.IDLE;
   573→  return BehaviorState.IDLE;
   574→}

   576→// ─── Agent ID Hashing (for pixel office engine) ───────────────
   577→
   578→const idToNumeric = new Map<string, number>();
   579→const numericToId = new Map<number, string>();
   580→let nextNumericId = 1;
   581→
   582→function hashAgentId(id: string): number {
   583→  const existing = idToNumeric.get(id);
   584→  if (existing !== undefined) return existing;
   585→  const num = nextNumericId++;
   586→  idToNumeric.set(id, num);
   587→  numericToId.set(num, id);
   588→  return num;
   589→}
   590→
   591→function unhashAgentId(num: number): string | undefined {
   592→  return numericToId.get(num);
   593→}
