   262→// ─── Pixel Office Layout (40×36) ─────────────────────────────
   263→const COLS = 40;
   264→const ROWS = 36;
   265→
   266→function buildTiles(): TileTypeVal[] {
   267→  const W = 0 as TileTypeVal;
   268→  const F1 = 1 as TileTypeVal;
   269→  const F2 = 2 as TileTypeVal;
   270→  const F3 = 3 as TileTypeVal;
   271→  const F4 = 4 as TileTypeVal;
   272→  const F5 = 5 as TileTypeVal;
   273→  const F6 = 6 as TileTypeVal;
   274→  const F7 = 7 as TileTypeVal;
   275→  const F8 = 8 as TileTypeVal;
   276→  const F9 = 9 as TileTypeVal;
   277→  const F10 = 10 as TileTypeVal;
   278→
   279→  const tiles: TileTypeVal[] = [];
   280→  for (let r = 0; r < ROWS; r++) {
   281→    for (let c = 0; c < COLS; c++) {
   282→      if (r === 0 || r === ROWS - 1 || c === 0 || c === COLS - 1) { tiles.push(W); continue; }
   283→      if (r === 10) { tiles.push((c >= 5 && c <= 6) || (c >= 18 && c <= 19) || (c >= 31 && c <= 32) ? F8 : W); continue; }
   284→      if (r === 20) { tiles.push((c >= 5 && c <= 6) || (c >= 18 && c <= 19) || (c >= 31 && c <= 32) ? F8 : W); continue; }
   285→      if (r === 30) { tiles.push((c >= 8 && c <= 9) || (c >= 28 && c <= 29) ? F8 : W); continue; }
   286→      if (r >= 1 && r <= 9) {
   287→        if (c === 13) { tiles.push(r >= 4 && r <= 5 ? F8 : W); continue; }
   288→        if (c === 25) { tiles.push(r >= 4 && r <= 5 ? F8 : W); continue; }
   289→        if (c >= 1 && c <= 12) { tiles.push(F1); continue; }
   290→        if (c >= 14 && c <= 24) { tiles.push(F2); continue; }
   291→        if (c >= 26 && c <= 38) { tiles.push(F3); continue; }
   292→        tiles.push(W); continue;
   293→      }
   294→      if (r >= 11 && r <= 19) {
   295→        if (c === 13) { tiles.push(r >= 14 && r <= 15 ? F8 : W); continue; }
   296→        if (c === 25) { tiles.push(r >= 14 && r <= 15 ? F8 : W); continue; }
   297→        if (c >= 1 && c <= 12) { tiles.push(F4); continue; }
   298→        if (c >= 14 && c <= 24) { tiles.push(F5); continue; }
   299→        if (c >= 26 && c <= 38) { tiles.push(F6); continue; }
   300→        tiles.push(W); continue;
   301→      }
   302→      if (r >= 21 && r <= 29) {
   303→        if (c === 13) { tiles.push(r >= 24 && r <= 25 ? F8 : W); continue; }
   304→        if (c === 25) { tiles.push(r >= 24 && r <= 25 ? F8 : W); continue; }
   305→        if (c >= 1 && c <= 12) { tiles.push(F9); continue; }
   306→        if (c >= 14 && c <= 24) { tiles.push(F10); continue; }
   307→        if (c >= 26 && c <= 38) { tiles.push(F7); continue; }
   308→        tiles.push(W); continue;
   309→      }
   310→      tiles.push(F7);
   311→    }
   312→  }
   313→  return tiles;
   314→}

   316→function buildTileColors(): Array<{ h: number; s: number; b: number; c: number } | null> {
   317→  const wallColor = { h: 214, s: 30, b: -100, c: -55 };
   318→  const cmdColor = { h: 30, s: 50, b: -43, c: -88 };
   319→  const meetColor = { h: 210, s: 35, b: -30, c: -75 };
   320→  const designColor = { h: 330, s: 40, b: -35, c: -80 };
   321→  const devColor = { h: 150, s: 40, b: -40, c: -82 };
   322→  const serverColor = { h: 200, s: 45, b: -50, c: -70 };
   323→  const researchColor = { h: 35, s: 55, b: -38, c: -85 };
   324→  const loungeColor = { h: 20, s: 60, b: -35, c: -85 };
   325→  const doorColor = { h: 35, s: 25, b: 10, c: 0 };
   326→  const qaColor = { h: 345, s: 40, b: -40, c: -82 };
   327→  const opsColor = { h: 25, s: 50, b: -38, c: -80 };
   328→  const tiles = buildTiles();
   329→  const colors: Array<{ h: number; s: number; b: number; c: number } | null> = [];
   330→  for (let i = 0; i < tiles.length; i++) {
   331→    const tile = tiles[i];
   332→    if (tile === 0) { colors.push(wallColor); continue; }
   333→    if (tile === 8) { colors.push(doorColor); continue; }
   334→    if (tile === 1) { colors.push(cmdColor); continue; }
   335→    if (tile === 2) { colors.push(meetColor); continue; }
   336→    if (tile === 3) { colors.push(designColor); continue; }
   337→    if (tile === 4) { colors.push(devColor); continue; }
   338→    if (tile === 5) { colors.push(serverColor); continue; }
   339→    if (tile === 6) { colors.push(researchColor); continue; }
   340→    if (tile === 7) { colors.push(loungeColor); continue; }
   341→    if (tile === 9) { colors.push(qaColor); continue; }
   342→    if (tile === 10) { colors.push(opsColor); continue; }
   343→    colors.push(wallColor);
   344→  }
   345→  return colors;
   346→}

   548→// ─── Pixel Office Status Mapping ──────────────────────────────

   550→function mapAgentStatusToActive(status: string): boolean {
   551→  const s = status?.toLowerCase() ?? 'idle';
   552→  return !['offline', 'idle', 'done'].includes(s);
   553→}
   554→
   555→function mapAgentStatusToTool(status: string): string | null {
   556→  const s = status?.toLowerCase() ?? '';
   557→  if (s.includes('thinking') || s.includes('working')) return 'Write';
   558→  if (s.includes('review') || s.includes('reading') || s.includes('waiting_api')) return 'Read';
   559→  return null;
   560→}
   561→
   562→function mapAgentStatusToBubble(status: string): 'permission' | 'waiting' | null {
   563→  const s = status?.toLowerCase() ?? '';
   564→  if (s.includes('waiting_approval')) return 'permission';
   565→  if (s.includes('waiting_api')) return 'waiting';
   566→  return null;
   567→}
   568→
   569→function mapAgentStatusToBehavior(status: string): BehaviorState {
   570→  const s = status?.toLowerCase() ?? 'idle';
   571→  if (s.includes('working') || s.includes('thinking')) return BehaviorState.WORKING;
   572→  if (s.includes('reviewing')) return BehaviorState.MEETING;
   573→  if (s.includes('waiting_approval')) return BehaviorState.MEETING;
   574→  if (s.includes('waiting_api')) return BehaviorState.RESEARCH;
   575→  if (s.includes('done')) return BehaviorState.IDLE;
   576→  return BehaviorState.IDLE;
   577→}

   579→// ─── Agent ID Hashing (for pixel office engine) ───────────────
   580→
   581→const idToNumeric = new Map<string, number>();
   582→const numericToId = new Map<number, string>();
   583→let nextNumericId = 1;
   584→
   585→function hashAgentId(id: string): number {
   586→  const existing = idToNumeric.get(id);
   587→  if (existing !== undefined) return existing;
   588→  const num = nextNumericId++;
   589→  idToNumeric.set(id, num);
   590→  numericToId.set(num, id);
   591→  return num;
   592→}
   593→
   594→function unhashAgentId(num: number): string | undefined {
   595→  return numericToId.get(num);
   596→}
