   262→// ─── Pixel Office Layout (40×21, 6 rooms in 3×2 grid) ────────
   263→const COLS = 40;
   264→const ROWS = 21;
   265→
   266→function buildTiles(): TileTypeVal[] {
   267→  const W = 0 as TileTypeVal;
   268→  const F1 = 1 as TileTypeVal; // Command Center
   269→  const F2 = 2 as TileTypeVal; // Meeting Room
   270→  const F3 = 3 as TileTypeVal; // Design Studio
   271→  const F4 = 4 as TileTypeVal; // Development
   272→  const F5 = 5 as TileTypeVal; // Server Room
   273→  const F6 = 6 as TileTypeVal; // Research Lab
   274→  const D = 8 as TileTypeVal;  // Doorway
   275→
   276→  const tiles: TileTypeVal[] = [];
   277→  for (let r = 0; r < ROWS; r++) {
   278→    for (let c = 0; c < COLS; c++) {
   279→      // Outer walls
   280→      if (r === 0 || r === ROWS - 1 || c === 0 || c === COLS - 1) { tiles.push(W); continue; }
   281→      // Horizontal interior wall (row 10) with doorways
   282→      if (r === 10) {
   283→        tiles.push(
   284→          (c >= 5 && c <= 6) || (c >= 18 && c <= 19) || (c >= 31 && c <= 32) ? D : W
   285→        );
   286→        continue;
   287→      }
   288→      // Vertical interior wall (col 13) with doorways
   289→      if (c === 13) {
   290→        tiles.push((r >= 4 && r <= 5) || (r >= 14 && r <= 15) ? D : W);
   291→        continue;
   292→      }
   293→      // Vertical interior wall (col 26) with doorways
   294→      if (c === 26) {
   295→        tiles.push((r >= 4 && r <= 5) || (r >= 14 && r <= 15) ? D : W);
   296→        continue;
   297→      }
   298→      // Room 1: Command Center (cols 1-12, rows 1-9)
   299→      if (r >= 1 && r <= 9 && c >= 1 && c <= 12) { tiles.push(F1); continue; }
   300→      // Room 2: Meeting Room (cols 14-25, rows 1-9)
   301→      if (r >= 1 && r <= 9 && c >= 14 && c <= 25) { tiles.push(F2); continue; }
   302→      // Room 3: Design Studio (cols 27-38, rows 1-9)
   303→      if (r >= 1 && r <= 9 && c >= 27 && c <= 38) { tiles.push(F3); continue; }
   304→      // Room 4: Development (cols 1-12, rows 11-19)
   305→      if (r >= 11 && r <= 19 && c >= 1 && c <= 12) { tiles.push(F4); continue; }
   306→      // Room 5: Server Room (cols 14-25, rows 11-19)
   307→      if (r >= 11 && r <= 19 && c >= 14 && c <= 25) { tiles.push(F5); continue; }
   308→      // Room 6: Research Lab (cols 27-38, rows 11-19)
   309→      if (r >= 11 && r <= 19 && c >= 27 && c <= 38) { tiles.push(F6); continue; }
   310→      tiles.push(W);
   311→    }
   312→  }
   313→  return tiles;
   314→}

   316→function buildTileColors(): Array<{ h: number; s: number; b: number; c: number } | null> {
   317→  const wallColor = { h: 214, s: 30, b: -100, c: -55 };
   318→  const cmdColor = { h: 30, s: 50, b: -43, c: -88 };     // warm beige
   319→  const meetColor = { h: 210, s: 35, b: -30, c: -75 };    // cool blue-gray
   320→  const designColor = { h: 330, s: 40, b: -35, c: -80 };  // pink-tinted
   321→  const devColor = { h: 150, s: 40, b: -40, c: -82 };     // green-tinted
   322→  const serverColor = { h: 200, s: 45, b: -50, c: -70 };  // steel blue
   323→  const researchColor = { h: 35, s: 55, b: -38, c: -85 }; // warm amber
   324→  const doorColor = { h: 35, s: 25, b: 10, c: 0 };
   325→  const tiles = buildTiles();
   326→  const colors: Array<{ h: number; s: number; b: number; c: number } | null> = [];
   327→  for (let i = 0; i < tiles.length; i++) {
   328→    const tile = tiles[i];
   329→    if (tile === 0) { colors.push(wallColor); continue; }
   330→    if (tile === 8) { colors.push(doorColor); continue; }
   331→    if (tile === 1) { colors.push(cmdColor); continue; }
   332→    if (tile === 2) { colors.push(meetColor); continue; }
   333→    if (tile === 3) { colors.push(designColor); continue; }
   334→    if (tile === 4) { colors.push(devColor); continue; }
   335→    if (tile === 5) { colors.push(serverColor); continue; }
   336→    if (tile === 6) { colors.push(researchColor); continue; }
   337→    colors.push(wallColor);
   338→  }
   339→  return colors;
   340→}

   539→function mapAgentStatusToActive(status: string): boolean {
   540→  const s = status?.toLowerCase() ?? 'idle';
   541→  return !['offline', 'idle', 'done'].includes(s);
   542→}
   543→
   544→function mapAgentStatusToTool(status: string): string | null {
   545→  const s = status?.toLowerCase() ?? '';
   546→  if (s.includes('thinking') || s.includes('working')) return 'Write';
   547→  if (s.includes('review') || s.includes('reading') || s.includes('waiting_api')) return 'Read';
   548→  return null;
   549→}
   550→
   551→function mapAgentStatusToBubble(status: string): 'permission' | 'waiting' | null {
   552→  const s = status?.toLowerCase() ?? '';
   553→  if (s.includes('waiting_approval')) return 'permission';
   554→  if (s.includes('waiting_api')) return 'waiting';
   555→  return null;
   556→}
   557→
   558→function mapAgentStatusToBehavior(status: string): BehaviorState {
   559→  const s = status?.toLowerCase() ?? 'idle';
   560→  if (s.includes('working') || s.includes('thinking')) return BehaviorState.WORKING;
   561→  if (s.includes('reviewing')) return BehaviorState.MEETING;
   562→  if (s.includes('waiting_approval')) return BehaviorState.MEETING;
   563→  if (s.includes('waiting_api')) return BehaviorState.RESEARCH;
   564→  if (s.includes('done')) return BehaviorState.IDLE;
   565→  return BehaviorState.IDLE;
   566→}

   568→// ─── Agent ID Hashing (for pixel office engine) ───────────────
   569→
   570→const idToNumeric = new Map<string, number>();
   571→const numericToId = new Map<number, string>();
   572→let nextNumericId = 1;
   573→
   574→function hashAgentId(id: string): number {
   575→  const existing = idToNumeric.get(id);
   576→  if (existing !== undefined) return existing;
   577→  const num = nextNumericId++;
   578→  idToNumeric.set(id, num);
   579→  numericToId.set(num, id);
   580→  return num;
   581→}
   582→
   583→function unhashAgentId(num: number): string | undefined {
   584→  return numericToId.get(num);
   585→}

   600→function DelegationReport({
   601→  tasks,
   602→  totalDurationMs,
   603→}: {
   604→  tasks: DelegatedTask[];
   605→  totalDurationMs: number;
   606→}) {
   607→  const [expanded, setExpanded] = useState(false);
   608→  const completedCount = tasks.filter((t) => t.status === 'completed').length;
   609→  const failedCount = tasks.filter((t) => t.status === 'failed').length;
   610→
   611→  return (
   612→    <div className="bg-slate-800/30 border border-slate-700/50 rounded-xl p-3 space-y-2">
   613→      <button
   614→        type="button"
   615→        onClick={() => setExpanded(!expanded)}
   616→        className="w-full flex items-center justify-between text-left"
   617→      >
   618→        <div className="flex items-center gap-2">
   619→          <span className="text-sm">🔄</span>
   620→          <span className="text-xs font-medium text-slate-300">
   621→            Delegated to {tasks.length} agent{tasks.length !== 1 ? 's' : ''}
   622→          </span>
   623→          <span className="flex items-center gap-1">
   624→            {completedCount > 0 && (
   625→              <Badge className="bg-emerald-500/20 text-emerald-300 text-[9px] h-4 px-1.5">✓ {completedCount}</Badge>
   626→            )}
   627→            {failedCount > 0 && (
   628→              <Badge className="bg-red-500/20 text-red-300 text-[9px] h-4 px-1.5">✗ {failedCount}</Badge>
   629→            )}
   630→          </span>
   631→        </div>
   632→        <div className="flex items-center gap-2">
   633→          <span className="text-[10px] text-slate-500">
   634→            {totalDurationMs > 0 ? `${(totalDurationMs / 1000).toFixed(1)}s` : ''}
   635→          </span>
   636→          {expanded ? <ChevronUp className="w-3.5 h-3.5 text-slate-500" /> : <ChevronDown className="w-3.5 h-3.5 text-slate-500" />}
   637→        </div>
   638→      </button>
   639→      <AnimatePresence>
   640→        {expanded && (
   641→          <motion.div
   642→            initial={{ height: 0, opacity: 0 }}
   643→            animate={{ height: 'auto', opacity: 1 }}
   644→            exit={{ height: 0, opacity: 0 }}
   645→            transition={{ duration: 0.2 }}
   646→            className="overflow-hidden"
   647→          >
   648→            <div className="space-y-1.5 pt-1">
   649→              {tasks.map((task, i) => (
   650→                <motion.div
   651→                  key={`${task.agentId}-${i}`}
   652→                  initial={{ opacity: 0, x: -10 }}
   653→                  animate={{ opacity: 1, x: 0 }}
   654→                  transition={{ delay: i * 0.05 }}
   655→                  className="flex items-start gap-2 p-2 rounded-lg bg-slate-800/40 border border-slate-700/30"
   656→                >
   657→                  <span className="mt-0.5">
   658→                    {task.status === 'completed' ? <CheckCircle2 className="w-3.5 h-3.5 text-emerald-400" />
   659→                      : task.status === 'failed' ? <XCircle className="w-3.5 h-3.5 text-red-400" />
   660→                      : <Loader2 className="w-3.5 h-3.5 text-yellow-400 animate-spin" />}
   661→                  </span>
   662→                  <div className="flex-1 min-w-0">
   663→                    <div className="flex items-center gap-1.5">
   664→                      <span className="text-xs font-medium text-slate-300">{task.agentName}</span>
   665→                      {task.durationMs !== undefined && (
   666→                        <span className="text-[10px] text-slate-500">{(task.durationMs / 1000).toFixed(1)}s</span>
   667→                      )}
   668→                    </div>
   669→                    <p className="text-[11px] text-slate-400 mt-0.5">{task.task}</p>
   670→                    {task.result && expanded && (
   671→                      <p className="text-[10px] text-slate-500 mt-1 line-clamp-3">{task.result}</p>
   672→                    )}
   673→                  </div>
   674→                </motion.div>
   675→              ))}
   676→            </div>
   677→          </motion.div>
   678→        )}
   679→      </AnimatePresence>
   680→    </div>
   681→  );
   682→}
